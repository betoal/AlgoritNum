      iniSeg = lee2.substring(1,2);
      iniAp = leea.substring(1,2);
      
      iniciales = initPrim.concat(initSeg); // la concatenación es la suma de dos cadenas se puede sustituir concat con + (se supone)
      iniciales = iniciales.concat(initAp); // initPrim+initSeg
      inicialesd = iniPrim.concat(iniSeg);
      inicialesd = inicialesd.concat(iniAp);
      
      System.out.println(iniciales);
      System.out.println(inicialesd);
     
    }

    static BufferedReader kb = new BufferedReader(new InputStreamReader(System.in));

  public static boolean itera(int vez) throws Exception {
    if (vez==0)
       return true;
    p.println("\nDesea correr otra simulacion?");
	p.println("\"S\" para continuar; otro para terminar...");
	Resp=kb.readLine().toUpperCase();
	if (Resp.equals("S"))
	   return true;
    //endIf

  }//endItera       

  public static void main(String[] args) throws Exception {
	int raiz,N;
	Random Aleat;
    int cuenta=0;
   	while (true){
        if (!itera(cuenta))
           return;
        //endIf
        cuenta++;
		while (true){
		    p.println("Deme la raiz (entera) del generador de numeros aleatorios:");
		    try{raiz=Integer.parseInt(kb.readLine());break;}
		    catch (Exception e1) {continue;}
		}//endWhile
		Aleat = new Random(raiz);
		while (true){
		    p.println("\nCuantos aleatorios desea simular?");
	  	    try {N=Integer.parseInt(kb.readLine());break;}
  		    catch (Exception e){continue;}
		}//endWhile
		float Numero[] = new float [N];
		float mu_x=0, sigma_x=0;
		p.println("Deme el nombre del archivo de datos destino:");
		String FName=kb.readLine();
		PrintStream FN=new PrintStream(new FileOutputStream(new File(FName)));
		for (int i=0;i<N;i++){

		    mu_x=mu_x+X;
		    Numero[i]=X;
		}//endFor
		mu_x=mu_x/N;
		sigma_x=0;
		for (int j=0;j<N;j++){
		    float dif=mu_x-Numero[j];
		    sigma_x=sigma_x+dif*dif;
		}//endFor
		sigma_x=sigma_x/N;
		sigma_x=(float)Math.sqrt(sigma_x);
		p.printf("mu_x:    %11.5f\r\n",mu_x);

		FN.printf("mu_x:    %11.5f\r\n",mu_x);

	}//endWhile
